此题为三维偏序的模板题

需要注意的点是模板的计算方法会遗漏掉部分完全一致的答案，因为此时对于一组完全相同的 i 和 j 我们只会记录 i < j 的那组解

那么很好解决 我们只需要在CDQ分治之前提前将这种情况下的答案记录下来即可

因为树状数组比较熟练了 所以我们直接写注释讲解本题
```cpp
#include<algorithm>
#include<bitset>
#include<climits>
#include<cmath>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iomanip>
#include<iostream>
#include<map>
#include<queue>
#include<set>
#include<string>
#include<unordered_map>
#include<unordered_set>
#include<vector>

using namespace std;

typedef long long ll;
constexpr int mod = 1e9 + 7;

constexpr int maxn = 2e5 + 10;
constexpr int N = 20;
constexpr int Base = 13;
constexpr int dx[] = { 0,0,1,-1 };
constexpr int dy[] = { 1,-1,0,0 };

int n, k, tr[maxn], ans[maxn];

struct node {
    int a, b, c, id;
} v[maxn];

void add(int x,int y) {
    for (int i = x; i <= k; i += lowbit(i))
        tr[i] += y;
}

int query(int x) {
    int res = 0;
    for (int i = x; i >= 1; i -= lowbit(i))
        res += tr[i];
    return res;
}

bool cmp1(node c1, node c2) {
    if (c1.a == c2.a) {
        if (c1.b == c2.b) return c1.c < c2.c;
        return c1.b < c2.b;
    }
    return c1.a < c2.a;
}

bool cmp2(node c1, node c2) {
    return c1.b < c2.b;
}

void merge(int l, int r) {
    //利用树状数组解决二维偏序问题
    int mid = (l + r) / 2;
    int i = l;
    //i 为当前区间的左半部分 j 为当前区间的右半部分
    vector<int> a;//为了处理完这个询问后就清空 所以我们要记录下来有多少个数被添加了
    for (int j = mid + 1; j <= r; j++) {
        while (i <= mid && v[i].b <= v[j].b) {//利用双指针 把 b 比当前的小的 c 值加进去
            add(v[i].c, 1);
            a.push_back(v[i].c);
            i++;
        }
        ans[v[j].id] += query(v[j].c);//去查询有多少比当前的 c 更小的
    }
    for (auto v : a)
        add(v, -1);//清空 防止影响后续
    sort(v + l, v + r + 1, cmp2);//结束后按照第二维排序
}

void CDQ(int l, int r) {
    if (l == r) return;
    int mid = (l + r) / 2;
    CDQ(l, mid);//采用归并分治的思想 先调用左半部分
    CDQ(mid + 1, r);//再调用右半部分
    merge(l, r);//然后左右合并 同时记录答案
}

void solve() {
    cin >> n >> k;
    for (int i = 1; i <= n; i++) {
        cin >> v[i].a >> v[i].b >> v[i].c;
        v[i].id = i;//将问题离线 记录标号
    }
    sort(v + 1, v + n + 1, cmp1);//第一次排序 按照 a 从小到大排序即可 a 相同时按照 b 从小到大即可
    int r = 1;
    for (int i = 1; i <= n; i++) {
        while ((r + 1 <= n) && v[r + 1].a == v[i].a && v[r + 1].b == v[i].b && v[r + 1].c == v[i].c)
            r++;
        if (v[r].a == v[i].a && v[r].b == v[i].b && v[r].c == v[i].c)
            ans[v[i].id] += r - i;//单独考虑 a b c三者全部相同的答案
    }
    CDQ(1, n);//第一次排序之后 a 已经有序了 将问题降维为了二维偏序问题 调用 CDQ 分治
    vector<int> f(n + 1);
    //从此开始计算答案 不再赘述
    for (int i = 1; i <= n; i++)
        f[ans[i]]++;
    for (int i = 0; i < n; i++)
        cout << f[i] << "\n";
}

signed main() { 
    ios::sync_with_stdio(false); cin.tie(nullptr);
    int T = 1;
    while (T--) {
        solve();
    }
    return 0;
}
```
